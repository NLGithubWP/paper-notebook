```go
func search(nums []int, target int) int {
	left := 0
	right := len(nums)-1
  // must == mid,  otherwise it's not log(n)
	for left+1 < right{
		mid := (left + right) / 2
		if nums[mid] == target{
			return mid
		}else if nums[mid] < target {
			left = mid
		}else{
			right = mid
		}
	}
	if nums[left] == target{
		return left
	}
    if nums[right] == target{
		return right
	}
	
	return -1
}
```
